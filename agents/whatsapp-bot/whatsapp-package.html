// ===== BantouMind AI - WhatsApp Agent ===== // Agent WhatsApp Automatique - Version 1.0 // Prêt à l'emploi - Zéro configuration complexe const fs = require('fs'); const path = require('path'); class WhatsAppAgent { constructor(config = {}) { this.config = { botName: config.botName || 'BantouMind Assistant', welcomeMessage: config.welcomeMessage || 'Bonjour ! 👋 Je suis l\'assistant automatique. Comment puis-je vous aider ?', autoReply: config.autoReply !== false, workingHours: config.workingHours || '24/7', language: config.language || 'fr', ...config }; this.database = this.loadDatabase(); this.messages = this.loadTemplates(); console.log('✅ Agent WhatsApp initialisé'); console.log(`📱 Bot: ${this.config.botName}`); console.log(`⏰ Mode: ${this.config.workingHours}`); } loadDatabase() { try { return JSON.parse(fs.readFileSync('./database.json', 'utf8')); } catch { return { contacts: [], conversations: [], stats: {} }; } } loadTemplates() { try { return JSON.parse(fs.readFileSync('./templates.json', 'utf8')); } catch { return this.getDefaultTemplates(); } } getDefaultTemplates() { return { welcome: [ "Bonjour ! 👋 Bienvenue chez {{business_name}}. Je suis votre assistant virtuel. Comment puis-je vous aider aujourd'hui ?", "👋 Salut ! Moi c'est {{bot_name}}, l'assistant IA de {{business_name}}. En quoi puis-je vous être utile ?" ], order: { received: "✅ Commande reçue ! Voici le récapitulatif :\n\n{{order_details}}\n\nNous vous contacterons sous 24h pour la livraison.", confirmed: "🎉 Votre commande est confirmée ! Numéro : #{{order_id}}", delivered: "📦 Votre commande a été livrée ! Merci de votre confiance 🙏" }, followup: [ "Bonjour ! 👋 Vous aviez demandé des informations sur {{product}}. Souhaitez-vous commander ?", "Petit rappel : votre devis #{{quote_id}} est toujours valable 📋" ], info: [ "Nos horaires : {{hours}}", "Nos coordonnées : {{contact}}", "Retrouvez notre catalogue ici : {{catalog_link}}" ] }; } handleIncomingMessage(message) { const response = this.generateResponse(message); this.logConversation(message.from, message.text, response); return response; } generateResponse(message) { const text = message.text?.toLowerCase() || ''; if (text.includes('bonjour') || text.includes('salut') || text.includes('hello')) { return this.getRandomTemplate('welcome'); } if (text.includes('commander') || text.includes('acheter') || text.includes('prix')) { return this.getOrderResponse(message); } if (text.includes('merci') || text.includes('merci beaucoup')) { return "Avec plaisir ! 😊 N'hésitez pas si vous avez d'autres questions."; } if (text.includes('contact') || text.includes('adresse') || text.includes('ou')) { return this.getInfoResponse(); } if (text.includes('catalogue') || text.includes('produit')) { return "📋 Voici notre catalogue : {{catalog_link}}\n\n1️⃣ Produit A - 15 000 FCFA\n2️⃣ Produit B - 25 000 FCFA\n3️⃣ Produit C - 50 000 FCFA\n\nSouhaitez-vous commander ?"; } return "Merci pour votre message ! Un conseiller vous répondra dans les plus brefs délais. ⏳\n\nEn attendant, voici ce que je peux faire pour vous :\n• 💬 Voir notre catalogue\n• 📋 Passer une commande\n• 📞 Être mis en relation avec un conseiller"; } getRandomTemplate(key) { const templates = this.messages[key]; if (Array.isArray(templates)) { return templates[Math.floor(Math.random() * templates.length)]; } return templates; } getOrderResponse(message) { const orderId = Math.floor(Math.random() * 10000); return `🛒 Super ! Pour passer commande, j'ai besoin de ces informations :\n\n1️⃣ Nom du produit souhaité\n2️⃣ Quantité\n3️⃣ Votre adresse de livraison\n4️⃣ Votre numéro de téléphone\n\nEnvoyez-moi ces infos et je prépare votre commande #${orderId} ! ✅`; } getInfoResponse() { return `📍 Nous contacter :\n📞 Téléphone : {{contact_phone}}\n📧 Email : {{contact_email}}\n📍 Adresse : {{contact_address}}\n\n⏰ Horaires : {{hours}}`; } logConversation(from, message, response) { this.database.conversations.push({ from, message, response, timestamp: new Date().toISOString(), language: this.config.language }); if (!this.database.stats[from]) { this.database.contacts.push({ number: from, firstContact: new Date().toISOString(), status: 'new' }); } this.database.stats.totalConversations = (this.database.stats.totalConversations || 0) + 1; this.saveDatabase(); } saveDatabase() { fs.writeFileSync('./database.json', JSON.stringify(this.database, null, 2)); } sendFollowUp(contactNumber) { const template = this.getRandomTemplate('followup'); console.log(`📤 Message de relance envoyé à ${contactNumber}`); return template; } getStats() { return { totalConversations: this.database.stats.totalConversations || 0, totalContacts: this.database.contacts.length, activeToday: this.database.conversations.filter( c => new Date(c.timestamp).toDateString() === new Date().toDateString() ).length, status: 'actif', uptime: '24/7' }; } generateReport() { const stats = this.getStats(); return ` 📊 RAPPORT D'ACTIVITÉ - ${this.config.botName} ═══════════════════════════════════ 📱 Conversations totales : ${stats.totalConversations} 👥 Contacts uniques : ${stats.totalContacts} 📈 Conversations aujourd'hui : ${stats.activeToday} ✅ Statut : ${stats.status} ⏰ Uptime : ${stats.uptime} ═══════════════════════════════════`; } } // ===== DÉMARRAGE ===== console.log('╔═══════════════════════════════════╗'); console.log('║ BantouMind AI - WhatsApp Agent ║'); console.log('║ Version 1.0.0 ║'); console.log('╚═══════════════════════════════════╝'); // Configurer votre agent ici const agent = new WhatsAppAgent({ botName: 'BantouMind Assistant', businessName: 'Votre Entreprise', welcomeMessage: 'Bonjour ! 👋 Bienvenue !', workingHours: '24/7', language: 'fr' }); // Démarrer l'agent console.log(agent.generateReport()); // Export pour intégration module.exports = WhatsAppAgent;

📋 Guide d'Installation Rapide

  1. Télécharge le dossier whatsapp-agent
  2. Installe Node.js sur ton ordinateur (gratuit sur nodejs.org)
  3. Double-clique sur start.bat (Windows) ou start.sh (Mac/Linux)
  4. Configure tes messages dans le fichier config.json
  5. L'agent est prêt ! Il gère automatiquement les messages
